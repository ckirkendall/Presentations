function myTestSample (){
  var mock-world = google.testing.createMethodMock(my.namespace ,"world")
  mock-world().$returns("test")
              .$times(2);
    
  mock-world.$replay()
    
  assertEquals("hello test", hello());
  mock-world.$verify();
}

var runner = new goog.testing.TestRunner();
var tcase = new goog.testing.TestCase();
var test = new goog.test.TestCase.Test("hello world test" myTestSample);

tcase.add(myTestSample);

function init(){
	runner.initialize(tcase);
	runner.execute();
}
